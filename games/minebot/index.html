const mineflayer = require('mineflayer'); // Replace with your server details const bot = mineflayer.createBot({ host: 'mc.h4ks.com', // server IP or hostname port: 65535, // server port username: 'Bot', // your bot's username // password: 'your_password', // Uncomment if you need to log in with a password }); bot.on('spawn', () => { console.log('Bot has spawned in the game!'); }); bot.on('chat', (username, message) => { if (username === bot.username) return; // Ignore messages from the bot itself console.log(`<${username}> ${message}`); if (message === 'hi') { bot.chat('Hello!'); } }); bot.on('error', (err) => { console.log('Error:', err); }); bot.on('end', () => { console.log('Bot has disconnected from the server.'); });
